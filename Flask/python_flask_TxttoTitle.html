app.py

from flask import Flask, render_template, request

app = Flask(__name__)


@app.route("/pvuv")
def pvuv():
    # read file
    data = []
    with open("./data/pvuv.txt") as fin:
        is_first_line =True
        for line in fin:
            if is_first_line:
                is_first_line = False
                continue
            line = line[:-1] #\n
            pdate, pv, uv = line.split("\t")
            data.append((pdate, pv, uv))
    return render_template("pvuv.html", data=data)
